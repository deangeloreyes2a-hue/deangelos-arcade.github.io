let stages = [
"",
" O ",
" O \n | ",
" O \n/| ",
" O \n/|\\",
" O \n/|\\\n/ ",
" O \n/|\\\n/ \\"
];

let words = ["ARCADE","DEANGELO","SHARK","PLANE","RACING"];
let word = words[Math.floor(Math.random()*words.length)];
let guessed = [], wrong = 0;

function draw() {
  document.getElementById("man").innerText = stages[wrong];
  document.getElementById("word").innerText =
    word.split("").map(l => guessed.includes(l)?l:"_").join(" ");
}

document.addEventListener("keydown", e => {
  let l = e.key.toUpperCase();
  if (!/^[A-Z]$/.test(l) || guessed.includes(l)) return;
  guessed.push(l);

  if (!word.includes(l)) wrong++;
  draw();

  if (wrong === stages.length - 1) end("YOU LOSE ‚ùå");
  if (!document.getElementById("word").innerText.includes("_")) end("YOU WIN üéâ");
});

function end(text) {
  document.getElementById("result").innerText = text;
  document.getElementById("over").style.display = "flex";
}
